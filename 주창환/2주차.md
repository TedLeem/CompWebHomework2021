###### JS
```JavaScript
let num = 0;

let up = function() {
    num = num + 1;
    document.querySelector("#number").innerHTML = num;
}

let down = function() {
    num = num - 1;
    document.querySelector("#number").innerHTML = num;
}

let reset = function() {
    num = 0;
    document.querySelector("#number").innerHTML = num;
}
```
